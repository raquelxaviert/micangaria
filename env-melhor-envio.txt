# Configuração do Melhor Envio - Sandbox e Produção

# === AMBIENTE ===
# Descomente a linha abaixo para forçar o uso do sandbox
# MELHOR_ENVIO_SANDBOX=true

# === TOKENS DO MELHOR ENVIO ===
# Token para o ambiente de PRODUÇÃO
MELHOR_ENVIO_TOKEN=seu_token_de_producao_aqui

# Token para o ambiente de SANDBOX (ambiente de testes)
MELHOR_ENVIO_SANDBOX_TOKEN=seu_token_de_sandbox_aqui

# === CONFIGURAÇÃO AUTOMÁTICA ===
# Se NODE_ENV=development, o sistema usará automaticamente o sandbox
# Se NODE_ENV=production, o sistema usará automaticamente a produção
# Para forçar o sandbox mesmo em produção, defina MELHOR_ENVIO_SANDBOX=true

# === COMO OBTER OS TOKENS ===
# 1. SANDBOX:
#    - Crie uma conta em: https://sandbox.melhorenvio.com.br/
#    - Cadastre um aplicativo e obtenha o token
#    - Saldo fictício de R$ 10.000,00 para testes
#
# 2. PRODUÇÃO:
#    - Crie uma conta em: https://melhorenvio.com.br/
#    - Cadastre um aplicativo e obtenha o token
#    - Necessário saldo real para envios

# === LIMITAÇÕES DO SANDBOX ===
# - Apenas Correios e Jadlog disponíveis
# - Transações aprovadas automaticamente após 5 minutos
# - Status modificados automaticamente após 15 minutos
# - Não é possível realizar envios reais
